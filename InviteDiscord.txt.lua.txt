-- Tween info for animations

    0.8, -- Duration
    Enum.EasingStyle.Back, -- Easing Style
    Enum.EasingDirection.Out, -- Easing Direction
    0, -- Repeat Count
    false, -- Reverses
    0 -- Delay
)

    0.5,
    Enum.EasingStyle.Back,
    Enum.EasingDirection.In,
    0,

)

-- Copy to clipboard function
local function copyToClipboard(text)
    if setclipboard then
        setclipboard(text)
        -- Visual feedback
        local originalText = G2L["Msg2_7"].Text
        G2L["Msg2_7"].Text = "COPIED TO CLIPBOARD!"
        G2L["Msg2_7"].TextColor3 = Color3.fromRGB(100, 255, 100)
        
        wait(1.5)
        G2L["Msg2_7"].Text = originalText
        G2L["Msg2_7"].TextColor3 = Color3.fromRGB(100, 255, 100)
    end
end

-- Dragging functionality
local function startDrag(input)
    if not animationComplete then return end
    
    isDragging = true
    dragStart = input.Position
    startPos = G2L["Frame_2"].Position
end

local function updateDrag(input)
    if isDragging then
        local delta = input.Position - dragStart
        G2L["Frame_2"].Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end

local function endDrag()
    isDragging = false
end

-- Input connections
G2L["Frame_2"].InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        startDrag(input)
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        updateDrag(input)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        endDrag()
    end
end)

-- Close button functionality with animation
G2L["Close_8"].MouseButton1Click:Connect(function()
    -- Close animation
    local closeTween = TweenService:Create(G2L["Frame_2"], closeTweenInfo, {
        Position = UDim2.new(0.5, -135, 1, 50),
        Rotation = 15,
        Size = UDim2.new(0, 0, 0, 0)
    })
    
    -- Fade out animation
    local fadeTween = TweenService:Create(G2L["Frame_2"], closeTweenInfo, {
        BackgroundTransparency = 1
    })
    
    -- Fade out all children
    for _, child in pairs(G2L["Frame_2"]:GetChildren()) do
        if child:IsA("GuiObject") then
            TweenService:Create(child, closeTweenInfo, {
                BackgroundTransparency = 1,
                TextTransparency = 1
            }):Play()
        end
    end
    
    closeTween:Play()
    fadeTween:Play()
    
    closeTween.Completed:Connect(function()
        G2L["ScreenGui_1"]:Destroy()
    end)
end)

-- Entry animation
spawn(function()
    wait(0.5) -- Small delay before showing
    
    -- Target position (center of screen)
    local targetPosition = UDim2.new(0.5, -135, 0.5, -72)
    
    -- Create the main entrance tween
    local entranceTween = TweenService:Create(G2L["Frame_2"], tweenInfo, {
        Position = targetPosition
    })
    
    -- Play entrance animation
    entranceTween:Play()
    
    -- Text animations (staggered)
    wait(0.3)
    
    -- Animate text elements with staggered timing
    local textElements = {
        G2L["DiscordInvite1_3"],
        G2L["Msg1_4"],
        G2L["TextLabel_6"],
        G2L["Msg2_7"]
    }
    
    for i, element in ipairs(textElements) do
        spawn(function()
            element.TextTransparency = 1
            wait(i * 0.1)
            TweenService:Create(element, TweenInfo.new(0.5, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
                TextTransparency = 0
            }):Play()
        end)
    end
    
    entranceTween.Completed:Connect(function()
        animationComplete = true
        -- Auto-copy the Discord link
        spawn(function()
            wait(1) -- Wait a bit before auto-copying
            copyToClipboard(discordLink)
        end)
    end)
end)

            local pulseTween2 = TweenService:Create(G2L["Msg2_7"], TweenInfo.new(1, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut), {
                BackgroundTransparency = 0.2

            pulseTween:Play()
            pulseTween.Completed:Connect(function()
                pulseTween2:Play()
            end)

        else

        end
    end
end)
